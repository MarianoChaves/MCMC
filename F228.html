Desenhando um gráfico em python:

O código em python usado na aula para resolver parte do exercício 1, você pode rodar o código no replit. Nesse código graficamos a força em função da massa \(m\). Segue o código abaixo 

                                
#Biblioteca de fazer gráficos 
import matplotlib.pyplot as plt 
#Biblioteca pra fazer gambiarra 
import numpy as np 

#Pontos do eixo x 
m_x = np.linspace(0,10,50) 
#Definindo minha força 
def Fm(m):
    G = 1#6.67e-11
    r = 1 M = 10
    return (G/r**2)*(-m**2+m*M)

#Criar uma figura 
plt.figure(figsize=(4,2.5)) 
#Desenhar o gráfico 
plt.plot(m_x, Fm(m_x)) 
#Escala do eixo x 
plt.xlim(0,10) 
#Nomes aos eixos 
plt.xlabel("m (M)") 
plt.ylabel("$G/r^2/M$") 

#Salvar figura 
plt.savefig("output.pdf")
#Mostrar figura 
plt.show()

Simulando três corpos em órbita circular:

No código que pode ser encontrado no replit, usamos MRUV e a lei da gravitação para simular a órbita de três estrelas com a massa do sol, posicionadas como um triângulo equilátero.

                                
#IMPORTANDO PACOTES 
import numpy as np 
import matplotlib matplotlib.use("Agg") 
import matplotlib.pyplot as plt from matplotlib.animation 
import FFMpegWriter 
#DEFINIR METADADOS DO FILME 
metadata = dict(title='Movie Test', artist='Matplotlib', comment='Movie support!') 
writer = FFMpegWriter(fps=15, metadata=metadata) 
#CRIAR FIGURE 
fig = plt.figure() dt = 1.154e+5 
#DEFININDO O QUE É UMA ESTRELA 
class Star(): 
    def __init__(self, pos_init, vel_init, mass): 
        self.pos = np.array(pos_init) 
        self.vel = np.array(vel_init) self.m = mass 

#FORÇAS ENTRE ESTRELAS 
def F_grav(star1, star2): 
    r_vec = star1.pos-star2.pos 
    r=np.linalg.norm(r_vec) 
    G=6.67e-11 r_hat = r_vec/r 
    return G*star1.m*star2.m/r**2*r_hat 

#FAZENDO O TEMPO PASSAR 
def star_update(star, force): 
    ac = force/star.m 
    star.pos = star.pos+star.vel*dt+ac*dt**2/2 
    star.vel = star.vel+ac*dt 
    return 0 

#DEFININDO AS ESTRELAS 
G = 6.67e-11 
d = 1e12 m_sol=2e32 #kilo-grama 
v = np.sqrt(G*m_sol/(d)) 
cos60 = np.cos(1.0472) 
sin60 = np.sin(1.0472) 
cos30 = np.cos(0.523599) 
sin30 = np.sin(0.523599) 
pos1 = [0,0,0] #metros 
pos2 = [d,0,0] #metros 
pos3 = [d*cos60,d*sin60,0] #metros 

vel1 = [-v*sin30,+v*cos30,0] #m/s 
vel2 = [-v*cos60,-v*sin60,0] #m/s 
vel3 = [v,0,0] #m/s 
#CRIANDO AS ESTRELAS 
star1 = Star(pos1, vel1, m_sol) 
star2 = Star(pos2, vel2, m_sol) 
star3 = Star(pos3, vel3, m_sol) 
#CRIAR FILME 
with writer.saving(fig, "tri_orbita.mp4", 200): 
    for i in range(200): 
    #DEFININDO OS LIMITES DA FIGURA 
    plt.xlim(-2e12, 2e12) 
    plt.ylim(-2e12, 2e12) 
    plt.xlabel("distância (m)") 
    plt.ylabel("distância (m)") 
    #CRIAR PLOT 
    l1, = plt.plot([], [], 'k-o', markersize=7) 
    l2, = plt.plot([], [], 'k-o', markersize=7) 
    l3, = plt.plot([], [], 'k-o', markersize=7) 
    #DESENHAR ESTRELAS 
    l1.set_data(star1.pos[0], star1.pos[1]) 
    l2.set_data(star2.pos[0], star2.pos[1]) 
    l3.set_data(star3.pos[0], star3.pos[1]) 
    #DESENHAR VELOCIDADES 
    v1 = plt.arrow(star1.pos[0], star1.pos[1], star1.vel[0]*dt*10, 
                    star1.vel[1]*dt*10, width=d/30, head_width=d/10, color="b") 
    v2 = plt.arrow(star2.pos[0], star2.pos[1], star2.vel[0]*dt*10, 
                    star2.vel[1]*dt*10, width=d/30, head_width=d/10, color="b") 
    v3 = plt.arrow(star3.pos[0], star3.pos[1], star3.vel[0]*dt*10, 
                    star3.vel[1]*dt*10, width=d/30, head_width=d/10, color="b") 
    #SALVAR FRAME 
    writer.grab_frame() 
    plt.cla() 
    #CALCULAR FORÇAS 
    force_in_1 = F_grav(star2,star1)+F_grav(star3,star1) 
    force_in_2 = F_grav(star1,star2)+F_grav(star3,star2) 
    force_in_3 = F_grav(star1,star3)+F_grav(star2,star3) 
    #ANDAR NO TEMPO 
    star_update(star1, force_in_1) 
    star_update(star2, force_in_2) 
    star_update(star3, force_in_3)
