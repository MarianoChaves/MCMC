#Biblioteca pra fazer gambiarra 
import numpy as np 

#Pontos do eixo x 
m_x = np.linspace(0,10,50) 
#Definindo minha força 
def Fm(m):
    G = 1#6.67e-11
    r = 1 M = 10
    return (G/r**2)*(-m**2+m*M)

#Criar uma figura 
plt.figure(figsize=(4,2.5)) 
#Desenhar o gráfico

#Escala do eixo x 
plt.xlim(0,10) 
#Nomes aos eixos 
plt.xlabel("m (M)") 
plt.ylabel("$G/r^2/M$")
